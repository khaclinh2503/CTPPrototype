def assign_random_loadout(players, skills_cfg, pendants_cfg, pets_cfg, rng=None):
    """D-16/D-34/D-41: Random assignment with replacement.
    D-20: SK_TEDDY and SK_GAY_NHU_Y mutually exclusive — re-roll on conflict.
    """
    import random as _rng
    if rng is None:
        rng = _rng
    for p in players:
        # Skills: 5 slots, with replacement, mutual exclusion D-20
        pool = [s.id for s in skills_cfg.skills if p.rank in s.rank_config or (p.rank == "R" and "S" in s.rank_config)]
        p.skills = []
        for _ in range(5):
            candidate = rng.choice(pool)
            # D-20 mutual exclusion
            if candidate == "SK_TEDDY" and "SK_GAY_NHU_Y" in p.skills:
                candidate = rng.choice([x for x in pool if x != "SK_TEDDY"])
            elif candidate == "SK_GAY_NHU_Y" and "SK_TEDDY" in p.skills:
                candidate = rng.choice([x for x in pool if x != "SK_GAY_NHU_Y"])
            p.skills.append(candidate)
        # Pendants: 3 slots, with replacement
        pendant_pool = [pt.id for pt in pendants_cfg.pendants]
        p.pendants = [rng.choice(pendant_pool) for _ in range(3)]
        p.pendant_rank = rng.choice(["B", "A", "S", "R", "SR"])
        # Pet: 1 slot, random tier
        pet = rng.choice(pets_cfg.pets)
        p.pet = pet.id
        p.pet_tier = rng.randint(1, 5)
        p.pet_stamina = pet.max_stamina

```python
import random
from ctp.core.models import Player
from ctp.core.events import GameEvent, EventType

class SkillEngine:
    def __init__(self, skill_configs, pendant_configs, pet_configs):
        self.skill_configs = {s.id: s for s in skill_configs.skills}
        self.pendant_configs = {p.id: p for p in pendant_configs.pendants}
        self.pet_configs = {p.id: p for p in pet_configs.pets}
        self._skill_handlers = {}    # skill_id -> handler_fn
        self._pendant_handlers = {}  # pendant_id -> handler_fn
        self._pet_handlers = {}      # pet_id -> handler_fn

    def register_skill(self, skill_id: str, handler):
        self._skill_handlers[skill_id] = handler

    def register_pendant(self, pendant_id: str, handler):
        self._pendant_handlers[pendant_id] = handler

    def register_pet(self, pet_id: str, handler):
        self._pet_handlers[pet_id] = handler

    def calc_rate(self, skill_cfg, player: Player) -> float:
        """D-05: rate_at_star = base_rate + (current_star - min_star) * chance"""
        rank = player.rank
        if rank == "R":
            rank = "S"  # D-03
        rc = skill_cfg.rank_config.get(rank)
        if rc is None:
            return 0.0  # D-04
        return rc.base_rate + (player.star - rc.min_star) * rc.chance

    def fire(self, trigger: str, player: Player, ctx: dict) -> list:
        """Fire all skill handlers for trigger. Returns list of events/results.
        D-51: skills_disabled_this_turn check.
        ctx must contain: board, players, is_player_turn (bool).
        """
        if player.skills_disabled_this_turn:
            return []
        results = []
        for skill_id in player.skills:
            cfg = self.skill_configs.get(skill_id)
            if cfg is None:
                continue
            triggers = cfg.trigger if isinstance(cfg.trigger, list) else [cfg.trigger]
            if trigger not in triggers:
                continue
            handler = self._skill_handlers.get(skill_id)
            if handler is None:
                continue
            # Rate check (D-06)
            if cfg.always_active:
                result = handler(player, ctx, cfg, self)
            else:
                rate = self.calc_rate(cfg, player)
                if random.randint(0, 99) < rate:
                    result = handler(player, ctx, cfg, self)
                else:
                    result = None
            if result is not None:
                results.append(result)
        return results

    def fire_pendants(self, trigger: str, player: Player, ctx: dict) -> list:
        """Fire all pendant handlers for trigger.
        Pendants use pendant_rank for rate lookup (D-31).
        """
        if player.skills_disabled_this_turn:
            return []
        results = []
        for pendant_id in player.pendants:
            cfg = self.pendant_configs.get(pendant_id)
            if cfg is None:
                continue
            if trigger not in cfg.triggers:
                continue
            handler = self._pendant_handlers.get(pendant_id)
            if handler is None:
                continue
            if cfg.always_active:
                result = handler(player, ctx, cfg, self)
            else:
                rank = player.pendant_rank
                rate = getattr(cfg.rank_rates, rank, 0)
                if random.randint(0, 99) < rate:
                    result = handler(player, ctx, cfg, self)
                else:
                    result = None
            if result is not None:
                results.append(result)
        return results

    def fire_pet(self, trigger: str, player: Player, ctx: dict):
        """Fire pet handler if trigger matches and stamina > 0 (D-40)."""
        if player.pet is None or player.pet_stamina <= 0:
            return None
        cfg = self.pet_configs.get(player.pet)
        if cfg is None or cfg.trigger != trigger:
            return None
        handler = self._pet_handlers.get(player.pet)
        if handler is None:
            return None
        rate = cfg.tier_rates[player.pet_tier - 1]
        if random.randint(0, 99) < rate:
            player.pet_stamina -= 1  # D-40: decrement on activation
            return handler(player, ctx, cfg, self)
        return None
```
