**2. Create ctp/skills/engine.py — SkillEngine class:**

```python
"""SkillEngine: trigger dispatch, rate calculation, and random assignment."""

import random
from dataclasses import dataclass
from typing import Any, Callable

from ctp.core.models import Player
from ctp.core.events import GameEvent, EventType
from ctp.config.schemas import SkillEntry, PendantEntry, PetEntry


# Type alias for handler functions
SkillHandler = Callable[[Player, dict[str, Any]], list[GameEvent]]

# Handler registries (populated by handler modules in Plan 02)
_SKILL_HANDLERS: dict[str, SkillHandler] = {}
_PENDANT_HANDLERS: dict[str, SkillHandler] = {}
_PET_HANDLERS: dict[str, SkillHandler] = {}


def register_skill(skill_id: str, handler: SkillHandler) -> None:
    """Register a skill handler function."""
    _SKILL_HANDLERS[skill_id] = handler

def register_pendant(pendant_id: str, handler: SkillHandler) -> None:
    """Register a pendant handler function."""
    _PENDANT_HANDLERS[pendant_id] = handler

def register_pet(pet_id: str, handler: SkillHandler) -> None:
    """Register a pet handler function."""
    _PET_HANDLERS[pet_id] = handler


def calc_skill_rate(skill: SkillEntry, player_rank: str, player_star: int) -> float:
    """Calculate activation rate for a skill per D-05.

    Formula: base_rate + (current_star - min_star) * chance
    R rank uses S config per D-03.
    Returns 0.0 if skill unavailable for this rank per D-04.
    """
    effective_rank = "S" if player_rank == "R" else player_rank
    rank_cfg = skill.rank_config.get(effective_rank)
    if rank_cfg is None:
        return 0.0  # D-04
    return rank_cfg.base_rate + (player_star - rank_cfg.min_star) * rank_cfg.chance


def check_activation(rate: float) -> bool:
    """Check if activation succeeds per D-06: random(0,100) < rate."""
    return random.randint(0, 100) < rate


class SkillEngine:
    """Dispatches skill/pendant/pet triggers for a set of players.

    Usage:
        engine = SkillEngine(skill_configs, pendant_configs, pet_configs)
        events = engine.fire("ON_ROLL_AFTER", player, ctx)
    """

    def __init__(
        self,
        skill_configs: dict[str, SkillEntry],
        pendant_configs: dict[str, PendantEntry],
        pet_configs: dict[str, PetEntry],
    ):
        self.skill_configs = skill_configs
        self.pendant_configs = pendant_configs
        self.pet_configs = pet_configs

    def fire(self, trigger: str, player: Player, ctx: dict[str, Any]) -> list[GameEvent]:
        """Fire all skill/pendant/pet handlers for given trigger point.

        Checks player.skills_disabled_this_turn before firing skills.
        Returns list of GameEvents produced.
        """
        events: list[GameEvent] = []

        # Fire skills (unless disabled by SK_LOC_XOAY)
        if not player.skills_disabled_this_turn:
            for skill_id in player.skills:
                cfg = self.skill_configs.get(skill_id)
                if cfg is None or cfg.trigger != trigger:
                    continue
                rate = calc_skill_rate(cfg, player.rank, player.star)
                if rate <= 0:
                    continue
                if check_activation(rate):
                    handler = _SKILL_HANDLERS.get(skill_id)
                    if handler:
                        result = handler(player, ctx)
                        events.extend(result)

        # Fire pendants (pendant triggers can be multi)
        for pendant_id in player.pendants:
            pcfg = self.pendant_configs.get(pendant_id)
            if pcfg is None or trigger not in pcfg.triggers:
                continue
            # Pendant rate: direct lookup by pendant_rank
            rate = pcfg.rank_rates.get(player.pendant_rank, 0.0)
            if rate <= 0:
                continue
            if check_activation(rate):
                handler = _PENDANT_HANDLERS.get(pendant_id)
                if handler:
                    # Pass rate_2 in ctx for dual-rate pendants
                    pendant_ctx = dict(ctx)
                    if pcfg.rank_rates_2:
                        pendant_ctx["rate_2"] = pcfg.rank_rates_2.get(player.pendant_rank, 0.0)
                    result = handler(player, pendant_ctx)
                    events.extend(result)

        # Fire pet
        self._fire_pet(trigger, player, ctx, events)

        return events

    def _fire_pet(
        self, trigger: str, player: Player,
        ctx: dict[str, Any], events: list[GameEvent]
    ) -> None:
        """Fire pet handler if applicable. Per D-40: check stamina > 0 first."""
        if player.pet is None or player.pet_stamina <= 0:
            return
        pcfg = self.pet_configs.get(player.pet)
        if pcfg is None or pcfg.trigger != trigger:
            return
        tier_idx = player.pet_tier - 1
        if tier_idx < 0 or tier_idx >= len(pcfg.tier_rates):
            return
        rate = pcfg.tier_rates[tier_idx]
        if check_activation(rate):
            player.pet_stamina -= 1  # D-40: deplete stamina on activation
            handler = _PET_HANDLERS.get(player.pet)
            if handler:
                events.extend(handler(player, ctx))

    def fire_for_opponents(
        self, trigger: str, owner: Player, opponents: list[Player],
        ctx: dict[str, Any]
    ) -> list[GameEvent]:
        """Fire reactive triggers for opponent-triggered events.

        Used for triggers like ON_OPPONENT_LAND_YOURS where the OWNER's
        skills/pendants/pets fire in response to an opponent's action.
        """
        events: list[GameEvent] = []
        # Fire owner's skills/pendants/pets with the trigger
        events.extend(self.fire(trigger, owner, ctx))
        return events


def assign_random_loadout(
    player: Player,
    skill_pool: list[str],
    pendant_pool: list[str],
    pet_pool: list[str],
    pet_configs: dict[str, PetEntry],
    rank_choices: list[str] | None = None,
    pendant_rank_choices: list[str] | None = None,
) -> None:
    """Assign random skills/pendants/pet to player at game start per D-16/D-34/D-41.

    - 5 skills from pool WITH replacement (D-18/D-19)
    - 3 pendants from pool WITH replacement (D-34)
    - 1 pet from pool, tier random 1-5 (D-41)
    - Rank random from choices if provided (D-02)
    - Pendant rank random from choices if provided (D-31)
    """
    if rank_choices:
        player.rank = random.choice(rank_choices)
        player.star = random.randint(1, 5)
    if pendant_rank_choices:
        player.pendant_rank = random.choice(pendant_rank_choices)

    # D-19: 5 skill slots, with replacement (D-18)
    if skill_pool:
        player.skills = [random.choice(skill_pool) for _ in range(5)]
    else:
        player.skills = []
```
