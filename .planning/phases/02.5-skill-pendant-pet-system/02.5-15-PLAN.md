**Helper: tiles_needed_to_win(player, board):**
Win conditions (any 1 of 3):
1. **3 color pairs:** Own both CITY tiles of same color for 3 different colors
2. **1 full side:** Own ALL CITY + Resort tiles on 1 board side
3. **All Resorts:** Own all Resort tiles on map

```python
def tiles_needed_to_win(player, board):
    """Return minimum tiles needed to complete any win condition.
    Used by PET_XI_CHO to check if opponent is 1 tile away from winning.
    """
    min_needed = float('inf')

    # Condition 1: 3 color pairs
    color_groups = {}  # color -> [positions owned, positions total]
    for tile in board.board:
        if tile.space_id == SpaceId.CITY:
            color = tile.opt
            if color not in color_groups:
                color_groups[color] = {"owned": 0, "total": 0}
            color_groups[color]["total"] += 1  # should be 2 per color
            if tile.owner_id == player.player_id:
                color_groups[color]["owned"] += 1
    complete_pairs = sum(1 for c in color_groups.values() if c["owned"] == c["total"])
    partial_pairs = [(c["total"] - c["owned"]) for c in color_groups.values() if 0 < c["owned"] < c["total"]]
    if complete_pairs >= 3:
        return 0  # already won
    needed_for_pairs = 3 - complete_pairs
    if partial_pairs:
        partial_pairs.sort()
        tiles_for_condition1 = sum(partial_pairs[:needed_for_pairs]) if len(partial_pairs) >= needed_for_pairs else float('inf')
        min_needed = min(min_needed, tiles_for_condition1)

    # Condition 2: 1 full side (4 sides of diamond board)
    sides = _get_board_sides(board)
    for side_positions in sides:
        side_tiles = [board.get_tile(p) for p in side_positions if board.get_tile(p).space_id in (SpaceId.CITY, SpaceId.RESORT)]
        needed = sum(1 for t in side_tiles if t.owner_id != player.player_id)
        min_needed = min(min_needed, needed)

    # Condition 3: All Resorts
    resorts = [t for t in board.board if t.space_id == SpaceId.RESORT]
    needed_resorts = sum(1 for r in resorts if r.owner_id != player.player_id)
    min_needed = min(min_needed, needed_resorts)

    return min_needed

def _get_board_sides(board):
    """Return list of 4 sides, each a list of tile positions."""
    # 32-tile diamond: sides are positions [2-9], [10-17], [18-25], [26-32,1]
    # Adjust based on actual board layout — corners at 1, 9, 17, 25
    return [
        list(range(2, 9)),    # side 1
        list(range(10, 17)),  # side 2
        list(range(18, 25)),  # side 3
        list(range(26, 33)) + [1] if 33 > 32 else list(range(26, 33)),  # side 4 wraps
    ]
    # NOTE: Verify exact positions with Board.json / board.get_row_non_corner_positions()
```
