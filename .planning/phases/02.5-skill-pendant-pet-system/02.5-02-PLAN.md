```python
def _count_complete_color_pairs(self, player: Player) -> int:
    """Count how many complete color groups player owns."""
    from collections import Counter
    colors_owned = Counter()
    colors_total = Counter()
    for tile in self.board.board:
        if tile.space_id == SpaceId.CITY:
            config = self.board.get_land_config(tile.opt)
            if config:
                color = config.get("color")
                colors_total[color] += 1
                if tile.owner_id == player.player_id:
                    colors_owned[color] += 1
    return sum(1 for c in colors_total if colors_owned.get(c, 0) == colors_total[c])
```
