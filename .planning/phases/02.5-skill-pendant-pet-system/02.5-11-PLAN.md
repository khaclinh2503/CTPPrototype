```python
def player_has_color_pair(player, board):
    """Check if player owns at least 2 CITY tiles of same color."""
    color_counts = {}
    for pos in player.owned_properties:
        tile = board.get_tile(pos)
        if tile.space_id == SpaceId.CITY:
            color_counts[tile.opt] = color_counts.get(tile.opt, 0) + 1
    return any(count >= 2 for count in color_counts.values())
```
