    ```python
    """Acquisition flow: player mua dat tu doi thu (forced sale)."""
    from ctp.core.models import Player
    from ctp.core.board import Board, Tile, SpaceId
    from ctp.core.constants import BASE_UNIT
    from ctp.core.events import EventBus, GameEvent, EventType

    def resolve_acquisition(
        player: Player,
        tile: Tile,
        board: Board,
        players: list[Player],
        event_bus: EventBus,
        acquire_rate: float = 1.0,
    ) -> list[GameEvent]:
        """Xu ly acquisition khi player dung o dat nguoi khac.

        Flow (per D-14):
        1. Chi apply cho CITY tiles co owner khac player
        2. Chi khi chua max level (building_level < 5)
        3. Tinh acquire_price = build cost level 1 * BASE_UNIT * acquire_rate
        4. Neu player du tien -> forced buy (B khong co quyen tu choi)
        5. Transfer: A pay -> B receive, tile.owner_id = A, update owned_properties

        Stub (per D-14): luon mua neu du tien.
        """
        events = []
        # Chi CITY tiles
        if tile.space_id != SpaceId.CITY:
            return events
        # Chi dat nguoi khac
        if tile.owner_id is None or tile.owner_id == player.player_id:
            return events
        # Chi khi chua max level
        if tile.building_level >= 5:
            return events

        land_config = board.get_land_config(tile.opt)
        if not land_config:
            return events

        # Gia mua = build cost level 1 * BASE_UNIT * acquireRate (per D-16)
        building = land_config.get("building", {})
        level_1_build = building.get("1", {}).get("build", 0)
        acquire_price = int(level_1_build * BASE_UNIT * acquire_rate)

        if not player.can_afford(acquire_price):
            return events

        # Stub: luon mua (per D-14)
        # Tim owner
        owner = next((p for p in players if p.player_id == tile.owner_id), None)
        if not owner:
            return events

        # Transfer
        player.cash -= acquire_price
        owner.receive(acquire_price)
        owner.remove_property(tile.position)
        player.add_property(tile.position)
        tile.owner_id = player.player_id

        events.append(GameEvent(
            event_type=EventType.PROPERTY_ACQUIRED,
            player_id=player.player_id,
            data={
                "position": tile.position,
                "from_player": owner.player_id,
                "price": acquire_price,
                "level": tile.building_level
            }
        ))
        event_bus.publish(events[-1])
        return events
    ```

    ```python
    """Upgrade logic: player nang cap dat sau moi turn."""
    from ctp.core.models import Player
    from ctp.core.board import Board, SpaceId
    from ctp.core.constants import BASE_UNIT
    from ctp.core.events import EventBus, GameEvent, EventType

    def resolve_upgrades(
        player: Player,
        board: Board,
        event_bus: EventBus,
    ) -> list[GameEvent]:
        """Upgrade tat ca properties cua player neu du tien.

        Stub (per D-15): luon upgrade neu du tien va chua max.
        Thu tu: upgrade theo thu tu trong owned_properties (Claude's discretion).
        Max level CITY = 5.
        """
        events = []
        for pos in list(player.owned_properties):  # copy list vi co the thay doi
            tile = board.get_tile(pos)
            if tile.space_id != SpaceId.CITY:
                continue
            if tile.building_level >= 5:
                continue

            land_config = board.get_land_config(tile.opt)
            if not land_config:
                continue

            next_level = tile.building_level + 1
            building = land_config.get("building", {})
            level_data = building.get(str(next_level), {})
            upgrade_cost = level_data.get("build", 0) * BASE_UNIT

            if player.can_afford(upgrade_cost):
                player.cash -= upgrade_cost
                tile.building_level = next_level

                events.append(GameEvent(
                    event_type=EventType.PROPERTY_UPGRADED,
                    player_id=player.player_id,
                    data={
                        "position": pos,
                        "new_level": next_level,
                        "cost": upgrade_cost
                    }
                ))
                event_bus.publish(events[-1])
        return events
    ```

    ```python
    """GameStrategy — MiniGame 3-round do den."""
    import random
    from ctp.tiles.base import TileStrategy
    from ctp.core.models import Player
    from ctp.core.board import Tile, Board
    from ctp.core.constants import STARTING_CASH
    from ctp.core.events import GameEvent, EventType

    class GameStrategy(TileStrategy):
        """MiniGame 3-round do den (per D-09 -> D-12).

        costOptions tu Board.json MiniGame: [0.05, 0.1, 0.15]
        maxChance: 3, increaseRate: 2
        Stub AI: luon chon muc min, dung sau round 1 neu thang.
        """
        DEFAULT_COST_OPTIONS = [0.05, 0.1, 0.15]
        DEFAULT_MAX_CHANCE = 3
        DEFAULT_INCREASE_RATE = 2

        def on_land(self, player, tile, board, event_bus, players=None):
            events = []
            cost_options = self.DEFAULT_COST_OPTIONS
            max_chance = self.DEFAULT_MAX_CHANCE
            increase_rate = self.DEFAULT_INCREASE_RATE

            # Stub AI: luon chon muc cuoc toi thieu (per D-12)
            bet_index = 0
            current_bet = int(cost_options[bet_index] * STARTING_CASH)  # 50,000

            total_winnings = 0
            rounds_played = 0

            for round_num in range(1, max_chance + 1):
                if round_num == 1:
                    # Round 1 bat buoc (per D-09)
                    player.cash -= current_bet
                    rounds_played = 1
                else:
                    # Stub: dung sau round 1 neu thang (per D-12)
                    break

                # 50% win
                if random.random() < 0.5:
                    # Thang: x(increase_rate^round_num)
                    multiplier = increase_rate ** round_num  # round1: x2, round2: x4, round3: x8
                    winnings = current_bet * multiplier
                    player.cash += winnings
                    total_winnings = winnings
                else:
                    # Thua: mat cuoc
                    total_winnings = -current_bet
                    break

            events.append(GameEvent(
                event_type=EventType.MINIGAME_RESULT,
                player_id=player.player_id,
                data={
                    "position": tile.position,
                    "rounds_played": rounds_played,
                    "bet": current_bet,
                    "result": total_winnings,
                    "won": total_winnings > 0
                }
            ))
            event_bus.publish(events[-1])
            return events

        def on_pass(self, player, tile, board, event_bus, players=None):
            return []
    ```
